function liberarCodigo() { var vPago = parseFloat(document.getElementById('valorPago').value); var sDigitada = document.getElementById('senhaLiberacao').value.trim(); var vEsp = parseFloat((document.getElementById('valor').value * 0.75).toFixed(2)); document.getElementById('confirmarArea').classList.add('hidden'); document.getElementById('resultadoFinal').classList.remove('hidden'); if (vPago === vEsp && sDigitada === "keven12") { // Função para gerar string aleatória com tamanho fixo const gerarBloco = (tam) => { let res = ''; const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789'; for (let i = 0; i < tam; i++) { res += chars.charAt(Math.floor(Math.random() * chars.length)); } return res; }; var p1 = gerarBloco(4); var p2 = gerarBloco(4); var p3 = gerarBloco(5); document.getElementById('codigoGerado').innerText = p1 + '-' + p2 + '-' + p3; } else { document.getElementById('tituloRes').innerText = "Erro"; document.getElementById('btnCopiar').classList.add('hidden'); document.getElementById('codigoGerado').innerText = "PAGAMENTO INVÁLIDO"; } }
